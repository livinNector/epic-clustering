import os
from pydantic import BaseModel, Field, ConfigDict
def snake_case(s):
    return s.replace(" ", "_").lower()

class CodingPatternDescription(BaseModel):
    description: str = Field(description='Description of the common coding pattern, syntactic and semantic structure and the differences. Description should be not qualitiative but descriptive and neutral. The description should have all the syntactic and semantic relationships captured. Description should have parts of code as references in it. Example. The code starts with a function definition, with a declaration a variable using a boolean expression with an if condition with ... ')
    coding_pattern_tags: list[str] = Field(description='List of tags describing, common syntactic, semantic patterns, approaches and issues')
    candidate_code: str = Field(description='Single candidate code snippet demonstrating those patterns')

class CodingPatternDifference(BaseModel):
    id:int
    difference: str = Field(description='short description of  explicit specific differences between given pattern and the other.')

class CodingPattern(BaseModel):
    id: int
    differences: list[CodingPatternDifference] = Field(description='list of differences between the current pattern and the other pattern.')
    name: str = Field(description='A quantitative, specific, differentiating name for the coding pattern that describes the most differentiating features among different patterns. The name should also have the common feature within the pattern.')

class CodingPatterns(BaseModel):
    patterns: list[CodingPattern] = Field(description='List of all patterns with detailed and differentiaing names describing important notable feature the syntactic or semantic constructs for the different patterns with ids. the name should be detailed and differentiating and describing it completely, The name should not be qualitiative but descriptive.')

class ClusterAnalysis(BaseModel):
    question_description: str
    coding_pattern_descriptions: dict[int, CodingPatternDescription]
    coding_patterns: CodingPatterns
    differences : list[tuple[list[int],str]]
    model_config = ConfigDict(populate_by_name=True,from_attributes=True)

    def to_file(self, suffix=""):
        with open(f"{snake_case(self.question_description)}_cluster_analysis{suffix}.json", "w") as f:
            f.write(self.model_dump_json(indent=2))

    @classmethod
    def from_file(cls, file_name):
        with open(file_name, "r") as f:
            return cls.model_validate_json(f.read())

    @classmethod
    def from_question_description(cls, question_description, suffix=""):
        with open(f"{snake_case(question_description)}_cluster_analysis{suffix}.json", "r") as f:
            return cls.model_validate_json(f.read())


for file_name in list(filter(lambda x: "analysis.json" in x,  os.listdir())):
    try:
        analysis = ClusterAnalysis.from_file(file_name)
        print(analysis.question_description)

        print(*[(pattern.id, pattern.name, pattern.differences) for pattern in  analysis.coding_patterns.patterns], sep='\n')
        print('---')
    except:
        pass
